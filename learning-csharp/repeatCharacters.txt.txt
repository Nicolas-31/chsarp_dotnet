  // do-while loop - RepeatCharacter
    public static void Main(string[] args)
    {
        Console.WriteLine("enter a character");
        var character = char.Parse(Console.ReadLine());
        Console.WriteLine("enter a target number");
        var number = int.Parse(Console.ReadLine());
        Console.WriteLine("The result is " + RepeatCharacter(character, number));
        Console.WriteLine("press something to exist the program");
        Console.ReadKey();
    }

    public static string RepeatCharacter(char character, int targetLength)
    {
        var result = "";
        do
        {
            result += character;
        } while (result.Length < targetLength);
        return result;
    }
